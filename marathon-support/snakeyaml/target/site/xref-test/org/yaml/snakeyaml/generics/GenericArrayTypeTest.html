62      public void testClasses() throws IntrospectionException {
63          GenericArray ga = new GenericArray();
64          Yaml yaml = new Yaml();
65          String doc = yaml.dump(ga);
66          // System.out.println(doc);
67          String etalon = "!!org.yaml.snakeyaml.generics.GenericArrayTypeTest$GenericArray\n"
68                  + "home: [1, 2, 3]\n" + "name: Array3\n";
69          assertEquals(etalon, doc);
70          if (JvmDetector.isProperIntrospection()) {
71              GenericArray parsed = (GenericArray) yaml.load(doc);
72              assertEquals("Array3", parsed.getName());
73              assertEquals(3, parsed.getHome().length);
74          } else {
75              try {
76                  yaml.load(doc);
77              } catch (Exception e) {
78                  // TODO Check GenericArrayType
79                  String message = "Cannot create property=home for JavaBean=org.yaml.snakeyaml.generics.GenericArrayTypeTest$GenericArray";
80                  assertTrue(e.getMessage(), e.getMessage().contains(message));
81              }
82          }
83      }

107     public void testJavaBean() throws IntrospectionException {
108         GenericArray ga = new GenericArray();
109         ArrayBean bean = new ArrayBean();
110         bean.setId("ID556677");
111         bean.setGa(ga);
112         JavaBeanDumper dumper = new JavaBeanDumper();
113         String doc = dumper.dump(bean);
114         // System.out.println(doc);
115         assertEquals(Util.getLocalResource("javabeans/genericArray-1.yaml"), doc);
116         //
117         JavaBeanLoader<ArrayBean> beanLoader = new JavaBeanLoader<ArrayBean>(ArrayBean.class);
118         if (JvmDetector.isProperIntrospection()) {
119             ArrayBean loaded = beanLoader.load(doc);
120             assertEquals("ID556677", loaded.getId());
121             assertEquals("Array3", loaded.getGa().getName());
122             assertEquals(3, loaded.getGa().getHome().length);
123         } else {
124             try {
125                 beanLoader.load(doc);
126             } catch (Exception e) {
127                 // TODO Check GenericArrayType
128                 String message = "Cannot create property=home for JavaBean=org.yaml.snakeyaml.generics.GenericArrayTypeTest$GenericArray";
129                 assertTrue(e.getMessage(), e.getMessage().contains(message));
130             }
131         }
132     }
