85      public void testChildrenArray() throws IOException {
86          Human_WithArrayOfChildren father = new Human_WithArrayOfChildren();
87          father.setName("Father");
88          father.setBirthday(new Date(1000000000));
89          father.setBirthPlace("Leningrad");
90          father.setBankAccountOwner(father);
91          //
92          Human_WithArrayOfChildren mother = new Human_WithArrayOfChildren();
93          mother.setName("Mother");
94          mother.setBirthday(new Date(100000000000L));
95          mother.setBirthPlace("Saint-Petersburg");
96          father.setPartner(mother);
97          mother.setPartner(father);
98          mother.setBankAccountOwner(father);
99          //
100         Human_WithArrayOfChildren son = new Human_WithArrayOfChildren();
101         son.setName("Son");
102         son.setBirthday(new Date(310000000000L));
103         son.setBirthPlace("Munich");
104         son.setBankAccountOwner(father);
105         son.setFather(father);
106         son.setMother(mother);
107         //
108         Human_WithArrayOfChildren daughter = new Human_WithArrayOfChildren();
109         daughter.setName("Daughter");
110         daughter.setBirthday(new Date(420000000000L));
111         daughter.setBirthPlace("New York");
112         daughter.setBankAccountOwner(father);
113         daughter.setFather(father);
114         daughter.setMother(mother);
115         //
116         Human_WithArrayOfChildren[] children = new Human_WithArrayOfChildren[] { son, daughter };
117         father.setChildren(children);
118         mother.setChildren(children);
119         //
120 
121         Constructor constructor = new Constructor(Human_WithArrayOfChildren.class);
122         TypeDescription HumanWithChildrenArrayDescription = new TypeDescription(
123                 Human_WithArrayOfChildren.class);
124         HumanWithChildrenArrayDescription.putListPropertyType("children",
125                 Human_WithArrayOfChildren.class);
126         constructor.addTypeDescription(HumanWithChildrenArrayDescription);
127 
128         Yaml yaml = new Yaml(constructor);
129         String output = yaml.dump(son);
130         // System.out.println(output);
131         String etalon = Util.getLocalResource("recursive/with-childrenArray.yaml");
132         assertEquals(etalon, output);
133         //
134         Human_WithArrayOfChildren son2 = (Human_WithArrayOfChildren) yaml.load(output);
135         assertNotNull(son2);
136         assertEquals("Son", son.getName());
137 
138         Human_WithArrayOfChildren father2 = son2.getFather();
139         assertEquals("Father", father2.getName());
140         assertEquals("Mother", son2.getMother().getName());
141         assertSame(father2, father2.getBankAccountOwner());
142         assertSame(father2.getPartner(), son2.getMother());
143         assertSame(father2, son2.getMother().getPartner());
144 
145         Human_WithArrayOfChildren[] fathersChildren = father2.getChildren();
146         assertEquals(2, fathersChildren.length);
147         Human_WithArrayOfChildren[] mothersChildren = father2.getPartner().getChildren();
148         assertEquals(2, mothersChildren.length);
149         assertSame(mothersChildren, fathersChildren);
150 
151         for (Object child : fathersChildren) {
152             // check if type descriptor was correct
153             assertSame(Human_WithArrayOfChildren.class, child.getClass());
154         }
155     }
