17  package org.yaml.snakeyaml.nodes;
18  
19  import org.yaml.snakeyaml.error.Mark;
20  
21  /***
22   * Base class for the two collection types {@link MappingNode mapping} and
23   * {@link SequenceNode collection}.
24   */
25  public abstract class CollectionNode extends Node {
26      private Boolean flowStyle;
27  
28      public CollectionNode(Tag tag, Mark startMark, Mark endMark, Boolean flowStyle) {
29          super(tag, startMark, endMark);
30          this.flowStyle = flowStyle;
31      }
32  
33      /***
34       * Serialization style of this collection.
35       * 
36       * @return <code>true</code> for flow style, <code>false</code> for block
37       *         style.
38       */
39      public Boolean getFlowStyle() {
40          return flowStyle;
41      }
42  
43      public void setFlowStyle(Boolean flowStyle) {
44          this.flowStyle = flowStyle;
45      }
46  
47      public void setEndMark(Mark endMark) {
48          this.endMark = endMark;
49      }
50  }
