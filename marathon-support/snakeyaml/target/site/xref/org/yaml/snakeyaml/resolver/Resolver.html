25  import org.yaml.snakeyaml.nodes.NodeId;
26  import org.yaml.snakeyaml.nodes.Tag;

28  /***
29   * Resolver tries to detect a type by scalars's content (when the type is
30   * implicit)
31   * 
32   * @see <a href="http://pyyaml.org/wiki/PyYAML">PyYAML</a> for more information
33   */
34  public class Resolver {
35      public static final Pattern BOOL = Pattern
36              .compile("^(?:yes|Yes|YES|no|No|NO|true|True|TRUE|false|False|FALSE|on|On|ON|off|Off|OFF)$");
37      public static final Pattern FLOAT = Pattern
38              .compile("^(?:[-+]?(?:[0-9][0-9_]*)//.[0-9_]*(?:[eE][-+][0-9]+)?|[-+]?(?:[0-9][0-9_]*)?//.[0-9_]+(?:[eE][-+][0-9]+)?|[-+]?[0-9][0-9_]*(?::[0-5]?[0-9])+//.[0-9_]*|[-+]?//.(?:inf|Inf|INF)|//.(?:nan|NaN|NAN))$");
39      public static final Pattern INT = Pattern
40              .compile("^(?:[-+]?0b[0-1_]+|[-+]?0[0-7_]+|[-+]?(?:0|[1-9][0-9_]*)|[-+]?0x[0-9a-fA-F_]+|[-+]?[1-9][0-9_]*(?::[0-5]?[0-9])+)$");
41      public static final Pattern MERGE = Pattern.compile("^(?:<<)$");
42      public static final Pattern NULL = Pattern.compile("^(?:~|null|Null|NULL| )$");
43      public static final Pattern EMPTY = Pattern.compile("^$");
44      public static final Pattern TIMESTAMP = Pattern
45              .compile("^(?:[0-9][0-9][0-9][0-9]-[0-9][0-9]-[0-9][0-9]|[0-9][0-9][0-9][0-9]-[0-9][0-9]?-[0-9][0-9]?(?:[Tt]|[ \t]+)[0-9][0-9]?:[0-9][0-9]:[0-9][0-9](?://.[0-9]*)?(?:[ \t]*(?:Z|[-+][0-9][0-9]?(?::[0-9][0-9])?))?)$");
46      public static final Pattern VALUE = Pattern.compile("^(?:=)$");
47      public static final Pattern YAML = Pattern.compile("^(?:!|&|//*)$");
48  
49      protected Map<Character, List<ResolverTuple>> yamlImplicitResolvers = new HashMap<Character, List<ResolverTuple>>();
50  
51      /***
52       * Create Resolver
53       * 
54       * @param respectDefaultImplicitScalars
55       *            false to parse/dump scalars as plain Strings
56       * @deprecated override addImplicitResolvers instead
57       */
58      public Resolver(boolean respectDefaultImplicitScalars) {
59          if (respectDefaultImplicitScalars) {
60              addImplicitResolvers();
61          }
62      }
63  
64      protected void addImplicitResolvers() {
65          addImplicitResolver(Tag.BOOL, BOOL, "yYnNtTfFoO");
66          addImplicitResolver(Tag.FLOAT, FLOAT, "-+0123456789.");
67          addImplicitResolver(Tag.INT, INT, "-+0123456789");
68          addImplicitResolver(Tag.MERGE, MERGE, "<");
69          addImplicitResolver(Tag.NULL, NULL, "~nN\0");
70          addImplicitResolver(Tag.NULL, EMPTY, null);
71          addImplicitResolver(Tag.TIMESTAMP, TIMESTAMP, "0123456789");
72          addImplicitResolver(Tag.VALUE, VALUE, "=");
73          // The following implicit resolver is only for documentation
74          // purposes.
75          // It cannot work
76          // because plain scalars cannot start with '!', '&', or '*'.
77          addImplicitResolver(Tag.YAML, YAML, "!&*");
78      }
79  
80      public Resolver() {
81          this(true);
82      }
83  
84      public void addImplicitResolver(Tag tag, Pattern regexp, String first) {
85          if (first == null) {
86              List<ResolverTuple> curr = yamlImplicitResolvers.get(null);
87              if (curr == null) {
88                  curr = new ArrayList<ResolverTuple>();
89                  yamlImplicitResolvers.put(null, curr);
90              }
91              curr.add(new ResolverTuple(tag, regexp));
92          } else {
93              char[] chrs = first.toCharArray();
94              for (int i = 0, j = chrs.length; i < j; i++) {
95                  Character theC = new Character(chrs[i]);
96                  if (theC == 0) {
97                      // special case: for null
98                      theC = null;
99                  }
100                 List<ResolverTuple> curr = yamlImplicitResolvers.get(theC);
101                 if (curr == null) {
102                     curr = new ArrayList<ResolverTuple>();
103                     yamlImplicitResolvers.put(theC, curr);
104                 }
105                 curr.add(new ResolverTuple(tag, regexp));
106             }
107         }
108     }
109 
110     public Tag resolve(NodeId kind, String value, boolean implicit) {
111         if (kind == NodeId.scalar && implicit) {
112             List<ResolverTuple> resolvers = null;
113             if ("".equals(value)) {
114                 resolvers = yamlImplicitResolvers.get('\0');
115             } else {
116                 resolvers = yamlImplicitResolvers.get(value.charAt(0));
117             }
118             if (resolvers != null) {
119                 for (ResolverTuple v : resolvers) {
120                     Tag tag = v.getTag();
121                     Pattern regexp = v.getRegexp();
122                     if (regexp.matcher(value).matches()) {
123                         return tag;
124                     }
125                 }
126             }
127             if (yamlImplicitResolvers.containsKey(null)) {
128                 for (ResolverTuple v : yamlImplicitResolvers.get(null)) {
129                     Tag tag = v.getTag();
130                     Pattern regexp = v.getRegexp();
131                     if (regexp.matcher(value).matches()) {
132                         return tag;
133                     }
134                 }
135             }
136         }
137         switch (kind) {
138         case scalar:
139             return Tag.STR;
140         case sequence:
141             return Tag.SEQ;
142         default:
143             return Tag.MAP;
144         }
145     }
