46      protected class RepresentJavaBean implements Represent {
47          public Node representData(Object data) {
48              try {
49                  return representJavaBean(getProperties(data.getClass()), data);
50              } catch (IntrospectionException e) {
51                  throw new YAMLException(e);
52              }
53          }

56      /***
57       * Tag logic:<br/>
58       * - explicit root tag is set in serializer <br/>
59       * - if there is a predefined class tag it is used<br/>
60       * - a global tag with class name is always used as tag. The JavaBean parent
61       * of the specified JavaBean may set another tag (tag:yaml.org,2002:map)
62       * when the property class is the same as runtime class
63       * 
64       * @param properties
65       *            JavaBean getters
66       * @param javaBean
67       *            instance for Node
68       * @return Node to get serialized
69       */
70      protected MappingNode representJavaBean(Set<Property> properties, Object javaBean) {
71          List<NodeTuple> value = new ArrayList<NodeTuple>(properties.size());
72          Tag tag;
73          Tag customTag = classTags.get(javaBean.getClass());
74          tag = customTag != null ? customTag : new Tag(javaBean.getClass());
75          // flow style will be chosen by BaseRepresenter
76          MappingNode node = new MappingNode(tag, value, null);
77          representedObjects.put(objectToRepresent, node);
78          boolean bestStyle = true;
79          for (Property property : properties) {
80              Object memberValue = property.get(javaBean);
81              NodeTuple tuple = representJavaBeanProperty(javaBean, property, memberValue, customTag);
82              if (tuple == null) {
83                  continue;
84              }
85              if (((ScalarNode) tuple.getKeyNode()).getStyle() != null) {
86                  bestStyle = false;
87              }
88              Node nodeValue = tuple.getValueNode();
89              if (!((nodeValue instanceof ScalarNode && ((ScalarNode) nodeValue).getStyle() == null))) {
90                  bestStyle = false;
91              }
92              value.add(tuple);
93          }
94          if (defaultFlowStyle != null) {
95              node.setFlowStyle(defaultFlowStyle);
96          } else {
97              node.setFlowStyle(bestStyle);
98          }
99          return node;
100     }

102     /***
103      * Represent one JavaBean property.
104      * 
105      * @param javaBean
106      *            - the instance to be represented
107      * @param property
108      *            - the property of the instance
109      * @param propertyValue
110      *            - value to be represented
111      * @param customTag
112      *            - user defined Tag
113      * @return NodeTuple to be used in a MappingNode. Return null to skip the
114      *         property
115      */
116     protected NodeTuple representJavaBeanProperty(Object javaBean, Property property,
117             Object propertyValue, Tag customTag) {
118         ScalarNode nodeKey = (ScalarNode) representData(property.getName());
119         boolean hasAlias = false;
120         if (this.representedObjects.containsKey(propertyValue)) {
121             // the first occurrence of the node must keep the tag
122             hasAlias = true;
123         }
124         Node nodeValue = representData(propertyValue);
125         // if possible try to avoid a global tag with a class name
126         if (nodeValue instanceof MappingNode && !hasAlias) {
127             // the node is a map, set or JavaBean
128             if (!Map.class.isAssignableFrom(propertyValue.getClass())) {
129                 // the node is set or JavaBean
130                 if (customTag == null && !nodeValue.getTag().equals(Tag.SET)) {
131                     // custom tag is not defined, set must keep the '!!set' tag
132                     if (property.getType() == propertyValue.getClass()) {
133                         // we do not need global tag because the property
134                         // Class is the same as the runtime class
135                         nodeValue.setTag(Tag.MAP);
136                     }
137                 }
138             }
139         } else if (propertyValue != null && Enum.class.isAssignableFrom(propertyValue.getClass())) {
140             nodeValue.setTag(Tag.STR);
141         }
142         if (nodeValue.getNodeId() != NodeId.scalar && !hasAlias) {
143             // generic collections
144             checkGlobalTag(property, nodeValue, propertyValue);
145         }
146         return new NodeTuple(nodeKey, nodeValue);
147     }

149     /***
150      * Remove redundant global tag for a type safe (generic) collection if it is
151      * the same as defined by the JavaBean property
152      * 
153      * @param property
154      *            - JavaBean property
155      * @param node
156      *            - representation of the property
157      * @param object
158      *            - instance represented by the node
159      */
160     @SuppressWarnings("unchecked")
161     protected void checkGlobalTag(Property property, Node node, Object object) {
162         Class<?>[] arguments = property.getActualTypeArguments();
163         if (arguments != null) {
164             if (node.getNodeId() == NodeId.sequence) {
165                 // apply map tag where class is the same
166                 Class<? extends Object> t = arguments[0];
167                 SequenceNode snode = (SequenceNode) node;
168                 Iterable<Object> memberList;
169                 if (object.getClass().isArray()) {
170                     memberList = Arrays.asList((Object[]) object);
171                 } else {
172                     // list
173                     memberList = (Iterable<Object>) object;
174                 }
175                 Iterator<Object> iter = memberList.iterator();
176                 for (Node childNode : snode.getValue()) {
177                     Object member = iter.next();
178                     if (member != null && t.equals(member.getClass())
179                             && childNode.getNodeId() == NodeId.mapping) {
180                         childNode.setTag(Tag.MAP);
181                     }
182                 }
183             } else if (object instanceof Set) {
184                 Class t = arguments[0];
185                 MappingNode mnode = (MappingNode) node;
186                 Iterator<NodeTuple> iter = mnode.getValue().iterator();
187                 Set set = (Set) object;
188                 for (Object member : set) {
189                     NodeTuple tuple = iter.next();
190                     if (t.equals(member.getClass())
191                             && tuple.getKeyNode().getNodeId() == NodeId.mapping) {
192                         tuple.getKeyNode().setTag(Tag.MAP);
193                     }
194                 }
195             } else if (node.getNodeId() == NodeId.mapping) {
196                 Class keyType = arguments[0];
197                 Class valueType = arguments[1];
198                 MappingNode mnode = (MappingNode) node;
199                 for (NodeTuple tuple : mnode.getValue()) {
200                     resetTag(keyType, tuple.getKeyNode());
201                     resetTag(valueType, tuple.getValueNode());
202                 }
203             }
204         }
205     }
206 
207     private void resetTag(Class<? extends Object> type, Node node) {
208         Tag tag = node.getTag();
209         if (tag.matches(type)) {
210             if (Enum.class.isAssignableFrom(type)) {
211                 node.setTag(Tag.STR);
212             } else {
213                 node.setTag(Tag.MAP);
214             }
215         }
216     }
